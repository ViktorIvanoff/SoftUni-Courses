function nextDay(year, month, day) {

    //    2016-10-1

    let nextDay = new Date(year, month - 1, day + 1);
    let newYear = nextDay.getFullYear();
    let newMonth = nextDay.getMonth() + 1;
    let newDate = nextDay.getDate();

    let result = `${newYear}-${newMonth}-${newDate}`;
    console.log(result);


}
